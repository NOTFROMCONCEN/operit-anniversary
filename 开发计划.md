### 2.2 工具导出方式

来源：[`extended_file_tools.js:152-172`](../_operit_ref/app/src/main/assets/packages/extended_file_tools.js:152)、[`worldbook_tools.ts:364-433`](../_operit_ref/examples/worldbook/src/packages/worldbook_tools.ts:364)

```js
// 模式 A（IIFE 封装，单文件 JS 推荐）：
const MyTools = (function () {
    async function add_anniversary(params) { /* ... */ return { success: true, data: ... }; }
    async function wrapToolExecution(func, params) {
        try { const r = await func(params); complete(r); }
        catch (e) { complete({ success: false, code: "...", message: e.message }); }
    }
    return { add_anniversary: (p) => wrapToolExecution(add_anniversary, p) };
})();
exports.add_anniversary = MyTools.add_anniversary;

// 模式 B（直接导出，更简洁）：
exports.add_anniversary = (params) => wrap(addAnniversary, params);
```

**关键机制**：
- 工具函数返回结果对象后，由包装器调用全局 **`complete(result)`** 把结果交还宿主
- 全局 `complete` / `Tools` / `ToolPkg` / `console` 均由宿主注入，无需 import
- 错误统一在 `wrap`/`wrapToolExecution` 中 catch 并 `complete({success:false, code, message})`

const ErrorCode = { INVALID_DATE, INVALID_TITLE, NOT_FOUND, DUPLICATE_TITLE,

function ok(data) { return { success: true, data }; }

### 5.4 日期工具模块（纯函数，无副作用）

按 [`example.md` 第 336–365 行](example.md:336) 实现，全部本地 Date，禁 UTC：

- `isValidYMD(date)` — 正则 `^\d{4}-\d{2}-\d{2}$` + 真实日历校验；`(m===2&&d===29)` 返回特殊标记触发 `UNSUPPORTED_LEAP_DAY`
- `parseYMD(date)` → `{y,m,d}`
- `toLocalDate(date)` → `new Date(y, m-1, d, 0,0,0,0)`
- `getTodayYMDInShanghai()` — 用 `+08:00` 偏移取当天 `YYYY-MM-DD`
- `diffDays(from, to)` — 本地午夜 Date 相减 ÷ 86400000，四舍五入
- `nextOccurrence(base, today)` — 周年递增
- `getAnniversaryStatus(base, today)` → `{daysSince, nextDate, daysUntilNext, isToday}`

**计算规则**（对齐 [`example.md` 第 233–256 行](example.md:233)）：
- `daysSince = diffDays(base, today)`：当天=0、未来=负、过去=正
- `nextDate`：`today <= base(同年)` → 当年 base；`today > base(同年)` → 次年 base；当天 → 当年 base
- `daysUntilNext = diffDays(today, nextDate)`
- `isToday = (daysSince === 0)`

### 5.5 存储层模块（async，套用官方模式）

```js
async function ensureDataFile() {
  await Tools.Files.mkdir(DATA_DIR, true);
  const ex = await Tools.Files.exists(DATA_FILE);
  if (!ex?.exists) await Tools.Files.write(DATA_FILE, "[]", false);
}
async function loadAnniversaries() {
  await ensureDataFile();
  const { content } = await Tools.Files.read(DATA_FILE);
  const parsed = JSON.parse(content || "[]");
  if (!Array.isArray(parsed)) throw fail("STORAGE_READ_FAILED", "数据文件损坏：非数组格式"); // 不覆盖
  return parsed;
}
async function saveAnniversaries(items) {
  if (!Array.isArray(items)) throw fail("STORAGE_WRITE_FAILED", "写入数据非数组");
  await Tools.Files.write(BACKUP_FILE, JSON.stringify(items, null, 2), false); // 先备份
  await Tools.Files.write(DATA_FILE, JSON.stringify(items, null, 2), false);
}
```

### 防护机制架构

```
用户调用工具
    │
    ▼
wrap() 包装器
    │
    ├─ verifyIntegrity() ──→ 水印自检（_W 哈希比对）
    │                        ├─ 通过 → 执行正常逻辑
    │                        └─ 失败 → makeDegradedResponse() → 返回版权警告
    │
    └─ 正常执行
         └─ createAnniversary() → stampAuthorFingerprint() → 写入 _afp 到磁盘
```

### 取证原理

- **功能性水印**：`_W` 常量参与数据指纹计算，不是装饰性注释。篡改后自检失败。
- **散布式哨兵**：`_G_S::anniv::*` 字符串散布于各源文件功能代码中，删除后哨兵检测失败。
- **数据级指纹**：每条记录写入 `_afp = hash(id + createdAt + salt)`。即使侵权者删除所有可见版权头，磁盘上的记录仍携带原始作者指纹。
- **差异检测**：侵权者复制代码移除防护模块后，新记录将缺失 `_afp` 字段，与原始记录形成可检测差异。
